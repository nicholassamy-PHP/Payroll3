  <script>
    let company = JSON.parse(localStorage.getItem('company') || '{}');
    let user = JSON.parse(localStorage.getItem('user') || '{}');
    let employees = [];
    let hoursData = {};

    document.addEventListener('DOMContentLoaded', () => {
      if (!user.id || !company.id) {
        window.location.href = '/dashboard.html';
        return;
      }
      loadDashboard();
      loadEmployees();
      loadReports();
    });

    function switchTab(tabName) {
      document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
      document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
      document.getElementById(tabName).classList.add('active');
      document.querySelector(`[onclick="switchTab('${tabName}')"]`).classList.add('active');
      
      if (tabName === 'payroll') loadEmployeesForPayroll();
      if (tabName === 'reports') loadReports();
    }

    function loadDashboard() {
      document.getElementById('currentYear').textContent = new Date().getFullYear();
      
      const companyHTML = `
        <p><strong>Company:</strong> ${company.company_name}</p>
        <p><strong>Type:</strong> ${company.company_type}</p>
        <p><strong>Address:</strong> ${company.company_address}</p>
        <p><strong>Industry:</strong> ${company.industry}</p>
      `;
      document.getElementById('companyInfo').innerHTML = companyHTML;
      
      fetch(`/api/payroll-employees?companyId=${company.id}`)
        .then(r => r.json())
        .then(data => {
          document.getElementById('totalEmployees').textContent = data.employees ? data.employees.length : 0;
        })
        .catch(e => console.log('Error loading employees for dashboard:', e));
    }

    function loadEmployees() {
      fetch(`/api/payroll-employees?companyId=${company.id}`)
        .then(r => r.json())
        .then(data => {
          employees = data.employees || [];
          renderEmployeeTable();
        })
        .catch(e => {
          showMessage('employeeMsg2', 'Error loading employees', 'error');
        });
    }

    function renderEmployeeTable() {
      const tbody = document.getElementById('employeeTable');
      if (!employees.length) {
        tbody.innerHTML = '<tr><td colspan="6" style="text-align: center; color: #64748b;">No employees yet. Add one above.</td></tr>';
        return;
      }

      tbody.innerHTML = employees.map(emp => `
        <tr>
          <td>${emp.first_name} ${emp.last_name}</td>
          <td>${emp.code}</td>
          <td>${emp.email}</td>
          <td>${emp.hire_date}</td>
          <td>${emp.active ? '✓ Active' : 'Inactive'}</td>
          <td>
            <button class="btn" style="padding: 6px 12px; font-size: 12px;" onclick="toggleEmployee('${emp.id}')">
              ${emp.active ? 'Deactivate' : 'Activate'}
            </button>
          </td>
        </tr>
      `).join('');
    }

    function addEmployee(e) {
      e.preventDefault();
      const msgDiv = document.getElementById('employeeMsg');
      msgDiv.style.display = 'none';

      const payload = {
        companyId: company.id,
        firstName: document.getElementById('firstName').value.trim(),
        lastName: document.getElementById('lastName').value.trim(),
        code: document.getElementById('employeeCode').value.trim(),
        email: document.getElementById('employeeEmail').value.trim(),
        hireDate: document.getElementById('hireDate').value
      };

      fetch('/api/payroll-employees', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      })
      .then(r => r.json())
      .then(data => {
        if (data.id) {
          showMessage('employeeMsg', '✓ Employee added successfully!', 'success');
          document.getElementById('employeeForm').reset();
          loadEmployees();
        } else {
          showMessage('employeeMsg', '✗ ' + (data.error || 'Failed to add employee'), 'error');
        }
      })
      .catch(e => showMessage('employeeMsg', '✗ Error: ' + e.message, 'error'));
    }

    function toggleEmployee(empId) {
      const emp = employees.find(e => e.id === empId);
      if (!emp) return;

      fetch('/api/payroll-employees', {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          id: empId,
          active: !emp.active
        })
      })
      .then(r => r.json())
      .then(data => {
        if (data.id) {
          loadEmployees();
        }
      })
      .catch(e => console.log('Error updating employee:', e));
    }

    function loadEmployeesForPayroll() {
      if (employees.length === 0) {
        document.getElementById('hoursContainer').innerHTML = '<p style="color: #64748b;">No employees. Add employees in the Employees tab first.</p>';
        return;
      }

      let html = '<div style="overflow-x: auto;"><table style="margin-top: 0;"><thead><tr>';
      html += '<th>Employee</th>';
      html += '<th>Regular</th><th>Holiday</th><th>Vacation</th><th>Maternity</th><th>SSL</th><th>Special</th><th>Other</th>';
      html += '</tr></thead><tbody>';

      employees.forEach(emp => {
        html += `<tr>
          <td style="font-weight: 600;">${emp.first_name} ${emp.last_name}</td>
          <td><input type="number" step="0.5" value="0" data-emp="${emp.id}" data-type="regular_hours"></td>
          <td><input type="number" step="0.5" value="0" data-emp="${emp.id}" data-type="holiday_paid_hours"></td>
          <td><input type="number" step="0.5" value="0" data-emp="${emp.id}" data-type="vacation_paid_hours"></td>
          <td><input type="number" step="0.5" value="0" data-emp="${emp.id}" data-type="maternity_hours"></td>
          <td><input type="number" step="0.5" value="0" data-emp="${emp.id}" data-type="ssl_hours"></td>
          <td><input type="number" step="0.5" value="0" data-emp="${emp.id}" data-type="special_hours"></td>
          <td><input type="number" step="0.01" value="0" data-emp="${emp.id}" data-type="other_amount"></td>
        </tr>`;
      });

      html += '</tbody></table></div>';
      document.getElementById('hoursContainer').innerHTML = html;
    }

    function calculatePayroll() {
      const msgDiv = document.getElementById('payrollMsg');
      msgDiv.style.display = 'none';

      const payEndDate = document.getElementById('payEndDate').value;
      const paymentDate = document.getElementById('paymentDate').value;

      if (!payEndDate || !paymentDate) {
        showMessage('payrollMsg', '✗ Please fill in dates', 'error');
        return;
      }

      const inputs = document.querySelectorAll('#hoursContainer input');
      const hoursMap = {};

      inputs.forEach(input => {
        const empId = input.dataset.emp;
        const type = input.dataset.type;
        const value = parseFloat(input.value) || 0;
        if (!hoursMap[empId]) hoursMap[empId] = {};
        hoursMap[empId][type] = value;
      });

      const payload = {
        companyId: company.id,
        payEndDate,
        paymentDate,
        hours: hoursMap
      };

      fetch('/api/payroll-calculate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      })
      .then(r => r.json())
      .then(data => {
        if (data.success) {
          showMessage('payrollMsg', '✓ Payroll calculated successfully!', 'success');
          displayPayrollResults(data.payroll);
        } else {
          showMessage('payrollMsg', '✗ ' + (data.error || 'Calculation failed'), 'error');
        }
      })
      .catch(e => showMessage('payrollMsg', '✗ Error: ' + e.message, 'error'));
    }

    function displayPayrollResults(payroll) {
      let html = '<table style="font-size: 14px;"><thead><tr>';
      html += '<th>Employee</th><th>Gross</th><th>CPP</th><th>EI</th><th>Federal Tax</th><th>Provincial Tax</th><th>Net Pay</th>';
      html += '</tr></thead><tbody>';

      if (Array.isArray(payroll)) {
        payroll.forEach(row => {
          html += `<tr>
            <td>${row.employee_name || 'Unknown'}</td>
            <td>$${parseFloat(row.gross_pay || 0).toFixed(2)}</td>
            <td>$${parseFloat(row.cpp_contribution || 0).toFixed(2)}</td>
            <td>$${parseFloat(row.ei_contribution || 0).toFixed(2)}</td>
            <td>$${parseFloat(row.federal_tax || 0).toFixed(2)}</td>
            <td>$${parseFloat(row.provincial_tax || 0).toFixed(2)}</td>
            <td style="font-weight: bold; color: #15803d;">$${parseFloat(row.net_pay || 0).toFixed(2)}</td>
          </tr>`;
        });
      }

      html += '</tbody></table>';
      document.getElementById('payrollResults').innerHTML = html;
      document.getElementById('resultsModal').classList.add('active');
    }

    function viewPayrollResults() {
      const modal = document.getElementById('resultsModal');
      if (modal.classList.contains('active')) {
        closeResultsModal();
      }
    }

    function closeResultsModal() {
      document.getElementById('resultsModal').classList.remove('active');
    }

    function loadReports() {
      fetch(`/api/payroll-employees?companyId=${company.id}`)
        .then(r => r.json())
        .then(data => {
          const tbody = document.getElementById('reportTable');
          const emps = data.employees || [];
          
          if (!emps.length) {
            tbody.innerHTML = '<tr><td colspan="7" style="text-align: center; color: #64748b;">No employees. Add employees to generate reports.</td></tr>';
            return;
          }

          tbody.innerHTML = emps.map(emp => `
            <tr>
              <td>${emp.first_name} ${emp.last_name}</td>
              <td>${emp.code}</td>
              <td>$${(emp.ytd_gross || 0).toFixed(2)}</td>
              <td>$${(emp.ytd_net || 0).toFixed(2)}</td>
              <td>$${(emp.ytd_cpp || 0).toFixed(2)}</td>
              <td>$${(emp.ytd_ei || 0).toFixed(2)}</td>
              <td>$${(emp.ytd_tax || 0).toFixed(2)}</td>
            </tr>
          `).join('');
        })
        .catch(e => {
          document.getElementById('reportTable').innerHTML = '<tr><td colspan="7" style="text-align: center; color: dc2626;">Error loading reports</td></tr>';
        });
    }

    function showMessage(elementId, msg, type) {
      const el = document.getElementById(elementId);
      el.textContent = msg;
      el.className = `message ${type}`;
      el.style.display = 'block';
    }

    function logout() {
      localStorage.clear();
      window.location.href = '/index.html';
    }

    window.onclick = (e) => {
      const modal = document.getElementById('resultsModal');
      if (e.target === modal) closeResultsModal();
    };
  </script>
