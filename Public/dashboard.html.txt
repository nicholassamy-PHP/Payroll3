  <script>
    function calculateTrialDaysRemaining(trialStartDate) {
      if (!trialStartDate) return 7;
      const startDate = new Date(trialStartDate);
      if (isNaN(startDate.getTime())) return 7;
      const today = new Date();
      const diffTime = startDate.getTime() + (7 * 24 * 60 * 60 * 1000) - today.getTime();
      const daysRemaining = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
      return Math.max(0, daysRemaining);
    }

    function updateTrialStatus() {
      const company = JSON.parse(localStorage.getItem('company') || '{}');
      const daysLeft = calculateTrialDaysRemaining(company.trial_start_date);

      const trialBanner = document.getElementById('trialBanner');
      const trialTitle = document.getElementById('trialTitle');
      const trialText = document.getElementById('trialText');
      const daysLeftEl = document.getElementById('daysLeft');
      const upgradeBtn = document.getElementById('upgradeBtn');

      if (daysLeft <= 0) {
        trialBanner.classList.add('expired');
        trialTitle.textContent = '⚠️ Trial Expired';
        trialText.textContent = 'Your free trial has ended. Please upgrade to continue using APEX PAY.';
        upgradeBtn.style.display = 'block';
        document.getElementById('accountStatus').textContent = 'Trial Expired - Upgrade Required';
      } else {
        daysLeftEl.textContent = daysLeft;
        upgradeBtn.style.display = 'block';
      }
    }

    function loadDashboard() {
      const user = JSON.parse(localStorage.getItem('user') || '{}');
      const company = JSON.parse(localStorage.getItem('company') || '{}');

      document.getElementById('userName').textContent = user.name || '-';
      document.getElementById('userEmail').textContent = user.email || '-';
      document.getElementById('companyName').textContent = company.company_name || '-';
      document.getElementById('companyType').textContent = company.company_type === 'single' ? 'Single Company' : 'Multi-Company';
      document.getElementById('companyAddress').textContent = company.company_address || '-';
      document.getElementById('industry').textContent = company.industry || '-';

      updateTrialStatus();
    }

    function logout() {
      localStorage.removeItem('authToken');
      localStorage.removeItem('user');
      localStorage.removeItem('company');
      window.location.href = '/';
    }

    function goToPayroll() {
      window.location.href = '/payroll';
    }

    function goToCheckout() {
      const company = JSON.parse(localStorage.getItem('company') || '{}');
      const user = JSON.parse(localStorage.getItem('user') || '{}');

      fetch('/api/stripe-checkout', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          userId: user.id,
          companyId: company.id,
          email: user.email
        })
      })
      .then(res => res.json())
      .then(data => {
        if (data.checkoutUrl) {
          window.location.href = data.checkoutUrl;
        } else {
          alert('Error: ' + (data.error || 'Could not create checkout'));
        }
      })
      .catch(err => alert('Error: ' + err.message));
    }

    loadDashboard();

    function openEditModal() {
      const company = JSON.parse(localStorage.getItem('company') || '{}');
      document.getElementById('editCompanyName').value = company.company_name || '';
      document.getElementById('editCompanyType').value = company.company_type || '';
      document.getElementById('editCompanyAddress').value = company.company_address || '';
      document.getElementById('editIndustry').value = company.industry || '';
      document.getElementById('editModal').style.display = 'block';
    }

    function closeEditModal() {
      document.getElementById('editModal').style.display = 'none';
    }

    window.onclick = function(event) {
      const modal = document.getElementById('editModal');
      if (event.target === modal) {
        modal.style.display = 'none';
      }
    }

    document.getElementById('editForm').addEventListener('submit', async (e) => {
      e.preventDefault();
      const company = JSON.parse(localStorage.getItem('company') || '{}');
      const companyName = document.getElementById('editCompanyName').value.trim();
      const companyType = document.getElementById('editCompanyType').value;
      const companyAddress = document.getElementById('editCompanyAddress').value.trim();
      const industry = document.getElementById('editIndustry').value;

      try {
        const response = await fetch('/api/update-company', {
          method: 'PUT',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            companyId: company.id,
            companyName,
            companyType,
            companyAddress,
            industry
          })
        });

        const data = await response.json();

        if (response.ok) {
          localStorage.setItem('company', JSON.stringify(data.company));
          loadDashboard();
          closeEditModal();
          alert('Company information updated successfully!');
        } else {
          alert('Error: ' + (data.error || 'Update failed'));
        }
      } catch (error) {
        alert('Error: ' + error.message);
      }
    });
